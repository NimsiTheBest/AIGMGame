  name:the game
  by:Nimsi
Layout
   use default
Buttons
*makemoney
name:Make money
desc:Click this button to make money!
on click:yield 1 money
class:bigButton
 
Resources
*money
name:Money
desc:This is how much money you got!

*superpoints
name:Super Points
desc:Super Points are used to buy Super buildings.

Buildings
*printer
name:Printer
desc:A printer that prints money. Illegal but profitable. 1 MPS.
cost:100 money
on tick:yield 1 money
tag:build, printer
 
*printerspawned
name:Spawned Printer
desc:This printer was spawned in. 1 MPS.
no buy
on tick:yield 1 money
tag:build, printer

*laptop
name:Laptop
desc:The same 3 videos on repeat. 11 MPS.
cost:1300 money
on tick:yield 11 money
tag:build, laptop

*superprinter
name:Super Printer
desc:This printer PRINTS printers! 25 MPS. Creates a free printer every 50 seconds.
cost:1000 money, 1 superpoint
on tick:yield 10 money
req:5 printer:max, supershack
tag:build, printer, super
 
Upgrades
*printupgrade1
name:More Ink
desc:Doubles the Money yield of Printers.
cost:500 money
passive:multiply money yield of tag:printer by 2
req:2 printer

*printupgrade2
name:Printer Stacking
desc:Stacks printers to be more effective. Every 10 printers, 3x yield for all Printers.
cost:500 money
passive:
	multiply money yield of tag:printer by max(1, floor(printer / 10))
end
req:10 printer
 
*printupgrade3
name:Even More Ink
desc:2.5x money from Printers
cost:2000 money
passive:
	multiply money yield of tag:printer by 2.5
end
req:15 printer, printupgrade1

*printupgrade4
name:Computer Wiring
desc:By connecting the Laptops to the Printers, we can sync them up. Printers gain 5% MPS from Laptops and vice versa.
cost:500 money
passive:
	multiply money yield of tag:printer by 1 + (laptop * 0.05)
	multiply money yield of tag:laptop by 1 + (printer * 0.05)
end
req:15 printer

*clickupgrade1
name:Click Glove
desc:+1 money per click.
cost:50 money
req:50 money:max
passive:increase money yield of makemoney by 1

*clickupgrade2
name:Metallic Fingertips
desc:+9 money per click.
cost:444 money
req:444 money:max
passive:increase money yield of makemoney by 9

*clickupgrade3
name:Extra Fingers
desc:+27 money per click.
cost:5555 money
req:5555 money:max
passive:increase money yield of makemoney by 27

*supershack
name:Super Shack
desc:Unlocks Super points. 0.01 SPPS.
cost:1 money
req:moneyach7
passive:show superpoints
on tick:yield 0.01 superpoints

*superupgrade1
name:Extra Lights
desc:Better lighting TRIPLES the MPS of all Super buildings.
cost:1 superpoint
req:2 superprinter
passive:multiply money yield of tag:super by 3
